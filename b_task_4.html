With the function you defined in func_drug_conc.R, run the drug concentration model with r2 taking values uniformly between 1/8 and 1/10, and plot the results.

source("./func/func_drug_conc.R")
r2_seq <- seq(1/10, 1/8, 0.005)
out_all <- NULL

for(ii in r2_seq){
    out <- deSolve::ode(
    y = c("x" = 1, "y" = 0), times = seq(1, 40, 0.01), func = func_drug_conc,
    parms = c("r1" = 1/2, "r2" = ii), method = "euler"
    )
    out_all <- cbind(out_all, out)

}
plot(x ~ time, data = out_all, type = "l")
colors <- c("red", "orange", "yellow", "green", "blue", "purple")
for(ii in 1:6){
    lines(x = out_all[,"time"], y = out_all[,ii*3], col = colors[ii] )
}
